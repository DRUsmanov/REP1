class Building {

private:
    int year;
    string type;

    void st_data(int y, string t) {
        year = y;
        type = t;
    }

    void get_info() {
        cout << "Type: " << type << ". Year:" << year << endl;
    }
};

	setlocale(LC_ALL, "RU");
